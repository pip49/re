#include <stdio.h>

int main() {
    int dan;

    printf("출력할 단을 입력하세요 (2-9): ");
    scanf("%d", &dan);

    if (dan < 2 || dan > 9) {
        printf("2에서 9 사이의 숫자만 입력해주세요.\n");
    } else {
        printf("--- %d단 ---\n", dan);
        for (int i = 1; i <= 9; i++) {
            printf("%d x %d = %d\n", dan, i, dan * i);
        }
    }

    return 0;
}
